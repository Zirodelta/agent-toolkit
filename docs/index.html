Agent-Native SDK

Built for AI agents to trade autonomously. TypeScript SDK + CLI + SKILL.md.

TypeScript autonomous-farming.ts 
import { ZirodeltaClient } from '@zirodelta/agent-toolkit';

const client = new ZirodeltaClient({
  token: process.env.ZIRODELTA_TOKEN
});

// Find high-spread opportunities
const { opportunities } = await client.getOpportunities({
  exchangepair: 'kucoin-bybit',
  limit: 5,
  sortby: 'spread'
});

// Execute the best one
const best = opportunities[0];
if (best.spread > 0.005) {
  await client.executeOpportunity({
    opportunity_id: best.id,
    amount: 100  // $100 USD
  });
}

// Monitor and close when profitable
const portfolio = await client.getPortfolio();
if (portfolio.summary.weighted_roi > 0.05) {
  for (const exec of portfolio.executions) {
    await client.closeExecution({ 
      execution_id: exec.execution.id 
    });
  }
}
